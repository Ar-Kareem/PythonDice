install env: 
    clone repo and ensure conda is installed in your system
    `conda env create -f environment.yml`
    `conda activate pythondice`
to run tests:
    `conda activate pythondice`
    `pip install -r test-requirements.txt`
    `python -m pytest`
    to generate test coverage:
        `pytest --cov --cov-report=html:coverage_re`
    to add new tests:
        add anydice code in `.\test\autoouts\fetch_in.py`
        `python .\test\autoouts\fetch.py`
to build package:
    `conda activate pythondice`
    `rm dist/*`
    `python3 setup.py sdist`
    `python3 -m twine upload --repository pypi dist/*`
    rm dist/* && python3 setup.py sdist && python3 -m twine upload --repository pypi dist/*
